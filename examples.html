Examples¶

Low Level API Example¶

This example shows how to use the low level API.

 1import sys
 2import time
 3
 4from toptica.lasersdk.client import Client, NetworkConnection
 5
 6
 7def my_callback(timestamp, name, value, exc):
 8    if exc:
 9        print('Error occurred!')
10        return
11
12    print('--------------------------------------')
13    print('Timestamp: ', timestamp)
14    print('     Name: ', name)
15    print('    Value: ', value)
16    print('--------------------------------------')
17
18
19def main():
20    with Client(NetworkConnection(sys.argv[1])) as client:
21        print("This is a '{}' with serial number '{}.'".format(
22            client.get('system-type', str),
23            client.get('serial-number', str)))
24
25        client.subscribe('laser1:dl:pressure-compensation:enabled', bool, my_callback)
26        client.set('laser1:dl:pressure-compensation:enabled', True)
27
28        time.sleep(1)
29
30
31if __name__ == "__main__":
32    main()

Synchronous API Example¶

This example shows the usage of the synchronous high level API.

 1import base64
 2import sys
 3import time
 4
 5from toptica.lasersdk.dlcpro.v2_0_3 import DLCpro, NetworkConnection, DeviceNotFoundError, DecopError, UserLevel
 6
 7script = 'OyBjaGVja3N1bSBmYzc4NDcwOWE5MTQwY2M1Mjg3ZWNiN2M1ZTAxYTQ3YQo7CihwYXJhbS1zZXQh' \
 8         'ICd1cHRpbWUgMTIzNCkKKGRpc3BsYXkgIkhhbGxpaGFsbG9cbiIp'
 9
10
11def my_callback(timestamp, name, value, exc):
12    if exc:
13        print('Error occurred!')
14        return
15
16    print('--------------------------------------')
17    print('Timestamp: ', timestamp)
18    print('     Name: ', name)
19    print('    Value: ', value)
20    print('--------------------------------------')
21
22
23def main():
24    try:
25        with DLCpro(NetworkConnection(sys.argv[1])) as dlc:
26            try:
27                print("\n\n=== Sync Read ===\n")
28                print('     System Time :', dlc.time.get())
29                print('          Uptime :', dlc.uptime.get())
30                print('Firmware Version :', dlc.fw_ver.get())
31                print('   System Health :', dlc.system_health_txt.get())
32                print('        Emission :', dlc.emission.get())
33                print('  Scan Frequency :', dlc.laser1.scan.frequency.get())
34
35                print("\n\n=== Sync Write ===\n")
36                old_label = dlc.system_label.get()
37                print('     System Label:', old_label)
38                dlc.system_label.set('::: THE LABEL :::')
39                print('     System Label:', dlc.system_label.get())
40                dlc.system_label.set(old_label)
41                print('     System Label:', dlc.system_label.get())
42
43                print("\n\n=== Sync Command Output ===\n")
44                print(dlc.system_summary())
45
46                print("\n\n=== Sync Command Output+Return ===\n")
47                dlc.change_ul(UserLevel.MAINTENANCE, 'CAUTION')
48                print('System Connections:', dlc.system_connections())
49                dlc.ul.set(3)
50
51                print("\n\n=== Sync Command Input ===\n")
52                print('           Uptime:', dlc.uptime.get())
53                dlc.service_script(base64.b64decode(script))
54                print('           Uptime:', dlc.uptime.get())
55
56                print("\n\n=== Sync Command Normal ===\n")
57                print(' User Level:', dlc.ul.get())
58                dlc.change_ul(UserLevel.MAINTENANCE, 'CAUTION')
59                print(' User Level:', dlc.ul.get())
60                dlc.ul.set(3)
61                print(' User Level:', dlc.ul.get())
62
63                print("\n\n=== Monitoring ===\n")
64                with dlc.uptime.subscribe(my_callback):
65                    for _ in range(5):
66                        dlc.poll()
67                        time.sleep(1)
68
69            except DecopError as error:
70                print(error)
71    except DeviceNotFoundError:
72        print('Device not found')
73
74
75if __name__ == "__main__":
76    main()

DLC pro Scope and Lock Example¶

This example shows how to read and display scope and lock data from a DLC pro scientific laser using Matplotlib.

  1import sys
  2
  3import matplotlib.pyplot as pyplot
  4
  5from toptica.lasersdk.dlcpro.v2_0_3 import DLCpro, NetworkConnection, DeviceNotFoundError
  6from toptica.lasersdk.utils.dlcpro import *
  7
  8
  9def main():
 10    try:
 11        with DLCpro(NetworkConnection(sys.argv[1])) as dlcpro:
 12            # Retrieve scan, lock raw data from device
 13            scope_data = extract_float_arrays('xyY', dlcpro.laser1.scope.data.get())
 14            raw_lock_candidates = dlcpro.laser1.dl.lock.candidates.get()
 15            lock_candidates = extract_lock_points('clt', raw_lock_candidates)
 16            lock_state = extract_lock_state(raw_lock_candidates)
 17
 18            # Create double y axis plot
 19            fig, laxis = pyplot.subplots()
 20            fig.suptitle('DLC pro Scope Output')
 21
 22            ch1_available = dlcpro.laser1.scope.channel1.signal.get() != -3  # Signal is 'none'
 23            ch2_available = dlcpro.laser1.scope.channel2.signal.get() != -3
 24
 25            # Set label and unit of X axis
 26            laxis.set_xlabel("{} [{}]".format(
 27                dlcpro.laser1.scope.channelx.name.get(),
 28                dlcpro.laser1.scope.channelx.unit.get()))
 29
 30            if ch1_available:
 31                red = laxis
 32
 33                # Set label and unit of left Y axis
 34                red.set_ylabel("{} [{}]".format(
 35                    dlcpro.laser1.scope.channel1.name.get(),
 36                    dlcpro.laser1.scope.channel1.unit.get()),
 37                    color='red')
 38
 39                # Plot first scope channel data
 40                red.plot(
 41                    scope_data['x'],
 42                    scope_data['y'],
 43                    linestyle='solid',
 44                    color='red',
 45                    zorder=1)
 46
 47                # Plot lock candidate points if available
 48                if 'c' in lock_candidates.keys():
 49                    red.plot(
 50                        lock_candidates['c']['x'],
 51                        lock_candidates['c']['y'],
 52                        linestyle='None',
 53                        marker='o',
 54                        markersize=12.0,
 55                        color='grey',
 56                        zorder=2)
 57
 58                # Plot selected lock candidate point if available
 59                if 'l' in lock_candidates.keys() and lock_state == 3:  # State is 'Selected'
 60                    red.plot(
 61                        lock_candidates['l']['x'],
 62                        lock_candidates['l']['y'],
 63                        linestyle='None',
 64                        marker='o',
 65                        markersize=16.0,
 66                        color='red',
 67                        markerfacecolor='none',
 68                        zorder=3)
 69
 70                # Retrieve and plot background trace data if lock is closed
 71                if lock_state == 5:  # State is 'Locked'
 72                    background_trace = extract_float_arrays('xy', dlcpro.laser1.dl.lock.background_trace.get())
 73
 74                    red.plot(
 75                        background_trace['x'],
 76                        background_trace['y'],
 77                        linestyle='solid',
 78                        color='lightgrey',
 79                        zorder=1)
 80
 81                # Plot lock tracking position if available
 82                if 't' in lock_candidates.keys():
 83                    red.plot(
 84                        lock_candidates['t']['x'],
 85                        lock_candidates['t']['y'],
 86                        linestyle='None',
 87                        marker='o',
 88                        markersize=20.0,
 89                        color='red',
 90                        markerfacecolor='none',
 91                        zorder=3)
 92
 93            # Plot second scope channel data if available
 94            if ch2_available:
 95                if ch1_available:
 96                    blue = laxis.twinx()
 97                else:
 98                    blue = laxis
 99
100                blue.set_ylabel("{} [{}]".format(
101                    dlcpro.laser1.scope.channel2.name.get(),
102                    dlcpro.laser1.scope.channel2.unit.get()),
103                    color='blue')
104
105                blue.plot(
106                    scope_data['x'],
107                    scope_data['Y'],
108                    linestyle='solid',
109                    color='blue',
110                    zorder=0)
111
112                laxis.set_zorder(blue.get_zorder() + 1)
113                laxis.patch.set_visible(False)
114
115            pyplot.margins(x=0.0)
116            pyplot.show()
117
118    except DeviceNotFoundError:
119        sys.stderr.write('Device not found')
120
121
122if __name__ == "__main__":
123    main()
